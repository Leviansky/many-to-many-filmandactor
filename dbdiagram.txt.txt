Table film {
  id integer [primary key]
  name string
  director string
  releaseYear integer
  company string
  actorId integer
  created_at timestamp
  updated_at timestamp 
}

Table actor {
  id integer [primary key]
  name string
  country string
  birthYear integer
  height integer
  filmId integer
  created_at timestamp
  updated_at timestamp
}

Table filmActors {
  id integer [primary key]
  actorId integer
  filmId integer
  created_at timestamp
  updated_at timestamp
}

Ref: film.id < filmActors.filmId
Ref: actor.id < filmActors.actorId
